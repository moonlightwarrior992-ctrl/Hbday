/* Fade Out */
function fadeOut(audio,duration){
let step=audio.volume/(duration/100);
let fade=setInterval(()=>{
if(audio.volume-step>0){
audio.volume-=step;
}else{
audio.pause();
clearInterval(fade);
}
},100);
}
